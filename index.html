      function CountAlph(STR,N)
      {
        var d = new Map();
       
        STR.split('').forEach(N => {
           
              if(d.has(N))
              {
                  d.set(N, d.get(N)+1);
              }
              else
                  d.set(N, 1);
        });
        STR.split('').forEach(N => {
          if(d.has(N) && d.get(N)!=0)
          {
            console.log( N +"="+ d.get(N) + "");
